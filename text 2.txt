import React, { useState, useEffect, useRef } from 'react';
import { 
  Sparkles, 
  BookOpen, 
  FileText, 
  HelpCircle, 
  Compass, 
  Send, 
  Upload, 
  CheckCircle2, 
  X, 
  Plus, 
  Trash2, 
  Award, 
  Moon, 
  ArrowRight,
  Info,
  ChevronRight,
  MessageSquare,
  RefreshCw,
  Zap,
  Target,
  Heart
} from 'lucide-react';

// Setup API Key (provided dynamically at runtime, defaults to empty)
const apiKey = "";

export default function App() {
  // Navigation
  const [activeTab, setActiveTab] = useState('hub'); // 'hub', 'quiz', 'dreams'

  // Input States
  const [inputText, setInputText] = useState('');
  const [imageFile, setImageFile] = useState(null);
  const [imagePreview, setImagePreview] = useState('');
  const [isProcessing, setIsProcessing] = useState(false);
  const [statusMessage, setStatusMessage] = useState('');

  // AI Content Output States
  const [summary, setSummary] = useState('');
  const [notes, setNotes] = useState('');
  
  // Quiz States
  const [quizQuestions, setQuizQuestions] = useState([]);
  const [selectedAnswers, setSelectedAnswers] = useState({});
  const [submittedQuiz, setSubmittedQuiz] = useState(false);
  const [quizScore, setQuizScore] = useState(0);

  // Chat States
  const [chatHistory, setChatHistory] = useState([
    { role: 'assistant', text: "Hello starfarer! 🌌 I am AstroAI. Drop some notes, text, or screenshots above, and we can explore the cosmos of knowledge together! Ask me any questions you have." }
  ]);
  const [chatInput, setChatInput] = useState('');
  const chatEndRef = useRef(null);

  // Dreams & Goals Board State
  const [dreams, setDreams] = useState([
    { id: 1, title: 'Ace Physics', content: 'Understand Special Relativity by next month!', category: 'Goal', color: 'purple', achieved: false },
    { id: 2, title: 'Build a Rocket', content: 'Study aerospace engineering at MIT!', category: 'Dream', color: 'indigo', achieved: false },
    { id: 3, title: 'Daily Mantra', content: 'Do not look back, you are not going that way. Keep sailing into the unknown.', category: 'Motivation', color: 'pink', achieved: true }
  ]);
  const [newDreamTitle, setNewDreamTitle] = useState('');
  const [newDreamContent, setNewDreamContent] = useState('');
  const [newDreamCategory, setNewDreamCategory] = useState('Goal');
  const [newDreamColor, setNewDreamColor] = useState('purple');

  // Custom Notifications/Modals
  const [toast, setToast] = useState({ show: false, message: '', type: 'success' });

  const showToast = (message, type = 'success') => {
    setToast({ show: true, message, type });
    setTimeout(() => setToast({ show: false, message: '', type: 'success' }), 4000);
  };

  // Scroll to bottom of chat
  useEffect(() => {
    chatEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  }, [chatHistory]);

  // Convert image to base64
  const handleImageChange = (e) => {
    const file = e.target.files[0];
    if (file) {
      if (file.size > 4 * 1024 * 1024) {
        showToast("Please choose an image under 4MB", "error");
        return;
      }
      setImageFile(file);
      const reader = new FileReader();
      reader.onloadend = () => {
        setImagePreview(reader.result);
        showToast("Cosmic scan ready for image analysis!", "success");
      };
      reader.readAsDataURL(file);
    }
  };

  const clearImage = () => {
    setImageFile(null);
    setImagePreview('');
  };

  // Exponential backoff API call helper
  const fetchWithRetry = async (url, options, maxRetries = 5) => {
    for (let i = 0; i < maxRetries; i++) {
      try {
        const response = await fetch(url, options);
        if (response.ok) {
          return await response.json();
        }
      } catch (err) {
        // Handled via exponential backoff silently
      }
      const delay = Math.pow(2, i) * 1000;
      await new Promise(res => setTimeout(res, delay));
    }
    throw new Error("Unable to establish steady telemetry with AstroAI. Please try again.");
  };

  // Generalized Gemini API Call
  const queryGemini = async (promptText, systemInstruction, imageBase64 = null) => {
    let url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
    
    let parts = [{ text: promptText }];
    if (imageBase64) {
      // Remove dataurl prefix if present
      const cleanBase64 = imageBase64.split(',')[1] || imageBase64;
      parts.push({
        inlineData: {
          mimeType: imageFile ? imageFile.type : "image/png",
          data: cleanBase64
        }
      });
    }

    const payload = {
      contents: [{ parts }],
      systemInstruction: { parts: [{ text: systemInstruction }] }
    };

    const options = {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(payload)
    };

    return await fetchWithRetry(url, options);
  };

  // Summarize Handler
  const generateSummary = async () => {
    if (!inputText && !imagePreview) {
      showToast("Please type text or upload an image first!", "warning");
      return;
    }

    setIsProcessing(true);
    setStatusMessage("Decrypting cosmic waves into summary...");

    const prompt = inputText 
      ? `Summarize this educational content in a structured, beautifully formatted layout. Use clear headings, bullet points, and highlight key terms: \n\n${inputText}`
      : "Analyze the uploaded image and extract the main content. Summarize it into clean, beautifully structured study notes using headings, bullet points, and highlight definitions.";

    const systemPrompt = "You are a professional, helpful educational summaries AI. Format your markdown strictly for standard display. Keep definitions precise, structure logical, and tone inspiring.";

    try {
      const data = await queryGemini(prompt, systemPrompt, imagePreview);
      const textOutput = data.candidates?.[0]?.content?.parts?.[0]?.text;
      if (textOutput) {
        setSummary(textOutput);
        showToast("Cosmic summary formulated!", "success");
      } else {
        throw new Error("Telemetric payload empty");
      }
    } catch (error) {
      showToast(error.message, "error");
    } finally {
      setIsProcessing(false);
      setStatusMessage('');
    }
  };

  // Notes Handler
  const generateNotes = async () => {
    if (!inputText && !imagePreview) {
      showToast("Please type text or upload an image first!", "warning");
      return;
    }

    setIsProcessing(true);
    setStatusMessage("Constructing comprehensive Star-Notes...");

    const prompt = inputText
      ? `Create deeply organized study notes from this text. Include: Introduction, Core Principles (with detailed explanations), Formulas/Keywords, Real-world analogies, and a Quick Review. Text:\n\n${inputText}`
      : "Analyze this image and design structured academic study notes from it. Organize with: Title, Core Concepts, In-depth Explanations, Key Takeaways, and a mnemonic/analogy helper.";

    const systemPrompt = "You are AstroNotes, a specialized digital note-taking AI. Create incredibly clean, professional, and dense study sheets with clear headers, lists, and visual spacing.";

    try {
      const data = await queryGemini(prompt, systemPrompt, imagePreview);
      const textOutput = data.candidates?.[0]?.content?.parts?.[0]?.text;
      if (textOutput) {
        setNotes(textOutput);
        showToast("Star-Notes drafted perfectly!", "success");
      } else {
        throw new Error("Telemetric payload empty");
      }
    } catch (error) {
      showToast(error.message, "error");
    } finally {
      setIsProcessing(false);
      setStatusMessage('');
    }
  };

  // Quiz Handler (uses Structured JSON)
  const generateQuiz = async () => {
    if (!inputText && !imagePreview) {
      showToast("Please provide study content (text or image) to forge a quiz!", "warning");
      return;
    }

    setIsProcessing(true);
    setStatusMessage("Forging quiz challenges...");

    const prompt = inputText
      ? `Create a challenging 5-question multiple choice quiz on the following material. Provide it in JSON format conforming strictly to the requested schema. Content:\n\n${inputText}`
      : `Analyze this educational image and create a 5-question multiple choice quiz based on its concepts. Provide it in JSON format conforming strictly to the requested schema.`;

    // Target API URL for Structured JSON
    const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
    
    let parts = [{ text: prompt }];
    if (imagePreview) {
      const cleanBase = imagePreview.split(',')[1] || imagePreview;
      parts.push({
        inlineData: {
          mimeType: imageFile ? imageFile.type : "image/png",
          data: cleanBase
        }
      });
    }

    const payload = {
      contents: [{ parts }],
      systemInstruction: { 
        parts: [{ text: "You are an expert examiner. Generate 5 highly relevant multiple choice questions on the provided study topic. Each must have 4 options, a correctIndex (0 to 3), and a comprehensive, encouraging explanation." }] 
      },
      generationConfig: {
        responseMimeType: "application/json",
        responseSchema: {
          type: "OBJECT",
          properties: {
            quiz: {
              type: "ARRAY",
              items: {
                type: "OBJECT",
                properties: {
                  question: { type: "STRING" },
                  options: {
                    type: "ARRAY",
                    items: { type: "STRING" }
                  },
                  correctIndex: { type: "INTEGER" },
                  explanation: { type: "STRING" }
                },
                required: ["question", "options", "correctIndex", "explanation"]
              }
            }
          },
          required: ["quiz"]
        }
      }
    };

    try {
      const response = await fetch(url, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(payload)
      });
      
      if (!response.ok) throw new Error("Cosmic network drop while forging quiz.");
      
      const resData = await response.json();
      const jsonText = resData.candidates?.[0]?.content?.parts?.[0]?.text;
      
      if (jsonText) {
        const parsed = JSON.parse(jsonText);
        if (parsed.quiz && parsed.quiz.length > 0) {
          setQuizQuestions(parsed.quiz);
          setSelectedAnswers({});
          setSubmittedQuiz(false);
          setQuizScore(0);
          setActiveTab('quiz'); // auto route to quiz tab
          showToast("A new Quantum Quiz has been forged!", "success");
        } else {
          throw new Error("Could not construct structured cosmic queries.");
        }
      } else {
        throw new Error("Telemetric payload empty");
      }
    } catch (error) {
      showToast(error.message, "error");
    } finally {
      setIsProcessing(false);
      setStatusMessage('');
    }
  };

  // Custom User Chat Prompt
  const handleChatSubmit = async (e) => {
    e.preventDefault();
    if (!chatInput.trim()) return;

    const userMessage = chatInput.trim();
    setChatHistory(prev => [...prev, { role: 'user', text: userMessage }]);
    setChatInput('');

    setIsProcessing(true);

    const contextText = `
      STUDY CONTEXT AVAILABLE:
      -----------------------
      Raw text: ${inputText}
      -----------------------
      Current summary: ${summary}
      -----------------------
      Current Notes: ${notes}
      -----------------------
    `;

    const systemPrompt = "You are AstroAI, a friendly educational mentor who is highly expert, encouraging, and witty. Answer the student's question accurately based on the context of their study materials. Give fun cosmic/space analogies when appropriate.";
    const userQuery = `Refer to this study context to answer my question. If the context does not contain the answer, use your scientific cosmic knowledge to guide me beautifully. Question: ${userMessage} \n\n${contextText}`;

    try {
      const data = await queryGemini(userQuery, systemPrompt);
      const textOutput = data.candidates?.[0]?.content?.parts?.[0]?.text;
      if (textOutput) {
        setChatHistory(prev => [...prev, { role: 'assistant', text: textOutput }]);
      } else {
        throw new Error("Failed to contact AstroAI base.");
      }
    } catch (error) {
      setChatHistory(prev => [...prev, { role: 'assistant', text: `⚠️ Error communicating with system: ${error.message}` }]);
    } finally {
      setIsProcessing(false);
    }
  };

  // Dreams/Goals Handlers
  const addDream = (e) => {
    e.preventDefault();
    if (!newDreamTitle.trim() || !newDreamContent.trim()) {
      showToast("Give your celestial dream a title and description!", "warning");
      return;
    }
    const created = {
      id: Date.now(),
      title: newDreamTitle.trim(),
      content: newDreamContent.trim(),
      category: newDreamCategory,
      color: newDreamColor,
      achieved: false
    };
    setDreams([created, ...dreams]);
    setNewDreamTitle('');
    setNewDreamContent('');
    showToast("Dream anchored in the stardust!", "success");
  };

  const toggleAchieved = (id) => {
    setDreams(dreams.map(d => {
      if (d.id === id) {
        const nextState = !d.achieved;
        if (nextState) showToast("Congratulations! Keep exploring! ⭐", "success");
        return { ...d, achieved: nextState };
      }
      return d;
    }));
  };

  const deleteDream = (id) => {
    setDreams(dreams.filter(d => d.id !== id));
    showToast("Dream orbit adjusted.", "info");
  };

  // Helper styles for colors
  const getColorClasses = (colorName, achieved) => {
    const opacity = achieved ? 'opacity-60 saturate-50' : 'opacity-100';
    switch (colorName) {
      case 'purple':
        return `bg-gradient-to-br from-violet-900/40 to-purple-900/40 border-purple-500/50 shadow-purple-900/30 text-purple-100 ${opacity}`;
      case 'indigo':
        return `bg-gradient-to-br from-indigo-900/40 to-blue-900/40 border-indigo-500/50 shadow-indigo-900/30 text-indigo-100 ${opacity}`;
      case 'pink':
        return `bg-gradient-to-br from-fuchsia-900/40 to-pink-900/40 border-pink-500/50 shadow-pink-900/30 text-pink-100 ${opacity}`;
      case 'gold':
        return `bg-gradient-to-br from-yellow-900/40 to-amber-900/40 border-amber-500/50 shadow-amber-900/30 text-amber-100 ${opacity}`;
      default:
        return `bg-slate-900/80 border-slate-700 text-slate-100 ${opacity}`;
    }
  };

  return (
    <div className="min-h-screen bg-slate-950 text-slate-100 font-sans selection:bg-purple-600 selection:text-white overflow-x-hidden relative pb-16">
      
      {/* Background stars overlay */}
      <div className="absolute inset-0 bg-[radial-gradient(ellipse_at_top,_var(--tw-gradient-stops))] from-purple-950/40 via-slate-950 to-black pointer-events-none z-0" />
      <div className="absolute top-20 left-10 w-72 h-72 bg-purple-600/10 rounded-full blur-3xl pointer-events-none" />
      <div className="absolute bottom-20 right-10 w-96 h-96 bg-indigo-600/10 rounded-full blur-3xl pointer-events-none" />

      {/* Floating stars */}
      <div className="absolute inset-0 opacity-20 pointer-events-none bg-[url('https://images.unsplash.com/photo-1506318137071-a8e063b4bec0?auto=format&fit=crop&w=1000&q=80')] mix-blend-color-dodge bg-cover bg-center z-0" />

      {/* Toast Alert */}
      {toast.show && (
        <div className={`fixed bottom-6 right-6 z-50 flex items-center space-x-2 px-5 py-3 rounded-xl border shadow-xl transition-all duration-300 transform translate-y-0 ${
          toast.type === 'success' ? 'bg-emerald-950/90 border-emerald-500 text-emerald-200' :
          toast.type === 'error' ? 'bg-rose-950/90 border-rose-500 text-rose-200' :
          'bg-indigo-950/90 border-indigo-500 text-indigo-200'
        }`}>
          <div className="w-2 h-2 rounded-full animate-ping bg-current" />
          <span className="font-medium text-sm">{toast.message}</span>
        </div>
      )}

      {/* Global Status Loader */}
      {isProcessing && (
        <div className="fixed inset-0 bg-slate-950/80 backdrop-blur-sm z-50 flex flex-col justify-center items-center">
          <div className="relative flex items-center justify-center">
            <div className="w-20 h-20 border-t-4 border-b-4 border-purple-500 rounded-full animate-spin"></div>
            <div className="absolute w-12 h-12 border-r-4 border-l-4 border-indigo-400 rounded-full animate-spin [animation-duration:1.5s]"></div>
            <Sparkles className="absolute text-purple-400 w-6 h-6 animate-pulse" />
          </div>
          <p className="mt-6 text-purple-200 text-lg font-medium tracking-wide animate-pulse">
            {statusMessage || "Syncing with NovaStudy matrix..."}
          </p>
        </div>
      )}

      {/* Navigation Header */}
      <header className="relative z-10 border-b border-purple-900/40 bg-slate-950/75 backdrop-blur-md sticky top-0 px-4 md:px-8 py-4">
        <div className="max-w-7xl mx-auto flex flex-col sm:flex-row justify-between items-center gap-4">
          
          {/* Logo */}
          <div className="flex items-center space-x-3 group">
            <div className="p-2.5 bg-gradient-to-tr from-purple-600 to-indigo-600 rounded-xl shadow-[0_0_15px_rgba(147,51,234,0.4)] transition-transform group-hover:scale-105">
              <Sparkles className="w-6 h-6 text-white animate-spin [animation-duration:12s]" />
            </div>
            <div>
              <h1 className="text-2xl font-bold tracking-tight bg-gradient-to-r from-purple-200 via-purple-400 to-indigo-300 bg-clip-text text-transparent">
                NovaStudy
              </h1>
              <span className="text-[10px] text-indigo-300 font-semibold uppercase tracking-widest block">
                Cosmic AI Study Room
              </span>
            </div>
          </div>

          {/* Navigation Controls */}
          <div className="flex bg-slate-900/90 border border-purple-950/60 p-1.5 rounded-xl space-x-1">
            <button
              onClick={() => setActiveTab('hub')}
              className={`flex items-center space-x-2 px-4 py-2 rounded-lg text-sm font-medium transition-all ${
                activeTab === 'hub'
                  ? 'bg-gradient-to-r from-purple-800 to-indigo-800 text-white shadow-lg'
                  : 'text-slate-400 hover:text-purple-300 hover:bg-purple-950/20'
              }`}
            >
              <Compass className="w-4 h-4" />
              <span>Cosmic Hub</span>
            </button>
            <button
              onClick={() => {
                if (quizQuestions.length === 0) {
                  showToast("Forge a quiz from materials in Cosmic Hub first!", "info");
                }
                setActiveTab('quiz');
              }}
              className={`flex items-center space-x-2 px-4 py-2 rounded-lg text-sm font-medium transition-all relative ${
                activeTab === 'quiz'
                  ? 'bg-gradient-to-r from-purple-800 to-indigo-800 text-white shadow-lg'
                  : 'text-slate-400 hover:text-purple-300 hover:bg-purple-950/20'
              }`}
            >
              <HelpCircle className="w-4 h-4" />
              <span>Quantum Quiz</span>
              {quizQuestions.length > 0 && !submittedQuiz && (
                <span className="absolute -top-1.5 -right-1.5 bg-pink-500 text-white text-[9px] px-1.5 py-0.5 rounded-full font-bold animate-bounce">
                  NEW
                </span>
              )}
            </button>
            <button
              onClick={() => setActiveTab('dreams')}
              className={`flex items-center space-x-2 px-4 py-2 rounded-lg text-sm font-medium transition-all ${
                activeTab === 'dreams'
                  ? 'bg-gradient-to-r from-purple-800 to-indigo-800 text-white shadow-lg'
                  : 'text-slate-400 hover:text-purple-300 hover:bg-purple-950/20'
              }`}
            >
              <Moon className="w-4 h-4" />
              <span>Nebula Board</span>
            </button>
          </div>
        </div>
      </header>

      {/* Main Container */}
      <main className="relative z-10 max-w-7xl mx-auto px-4 md:px-8 mt-8">
        
        {/* TAB 1: COSMIC HUB */}
        {activeTab === 'hub' && (
          <div className="grid grid-cols-1 lg:grid-cols-12 gap-8">
            
            {/* Input Column (Left Side) */}
            <div className="lg:col-span-5 space-y-6">
              
              {/* Cosmic Intake Station */}
              <div className="bg-slate-900/60 border border-purple-900/40 rounded-2xl p-6 backdrop-blur-md shadow-xl">
                <div className="flex items-center justify-between mb-4">
                  <div className="flex items-center space-x-2.5">
                    <div className="p-1.5 bg-purple-500/20 rounded-lg text-purple-400">
                      <BookOpen className="w-5 h-5" />
                    </div>
                    <h3 className="font-semibold text-lg text-purple-100">Intake Station</h3>
                  </div>
                  <span className="text-xs text-purple-400 font-mono">Telemetry Input</span>
                </div>

                <p className="text-xs text-slate-400 mb-4 leading-relaxed">
                  Provide textbook snippets, random notes, or upload a picture containing academic study content.
                </p>

                {/* Text input area */}
                <textarea
                  value={inputText}
                  onChange={(e) => setInputText(e.target.value)}
                  placeholder="Paste study text, lecture slide notes, raw transcript, or syllabus here..."
                  className="w-full h-44 bg-slate-950/80 border border-purple-950 text-slate-200 placeholder-slate-600 rounded-xl p-4 text-sm focus:outline-none focus:border-purple-500 focus:ring-1 focus:ring-purple-500 transition-all resize-none mb-4"
                />

                {/* Image Upload Area */}
                <div className="mb-6">
                  {!imagePreview ? (
                    <label className="flex flex-col items-center justify-center border-2 border-dashed border-purple-900/40 bg-slate-950/30 hover:bg-slate-950/60 rounded-xl p-6 cursor-pointer group transition-all">
                      <Upload className="w-8 h-8 text-purple-400 group-hover:text-purple-300 mb-2 transition-transform group-hover:-translate-y-0.5" />
                      <span className="text-xs font-semibold text-purple-300">Scan Textbook or Notes Image</span>
                      <span className="text-[10px] text-slate-500 mt-1">PNG, JPG up to 4MB</span>
                      <input 
                        type="file" 
                        accept="image/*" 
                        onChange={handleImageChange} 
                        className="hidden" 
                      />
                    </label>
                  ) : (
                    <div className="relative border border-purple-800 rounded-xl overflow-hidden bg-slate-950">
                      <img 
                        src={imagePreview} 
                        alt="Cosmic Scan Source" 
                        className="w-full h-32 object-cover opacity-70"
                      />
                      <div className="absolute inset-0 bg-gradient-to-t from-slate-950 via-transparent to-transparent flex items-end p-3">
                        <div className="flex items-center space-x-2 text-xs text-purple-200 font-mono bg-slate-950/80 px-2 py-1 rounded">
                          <CheckCircle2 className="w-3.5 h-3.5 text-emerald-400" />
                          <span>Image uploaded</span>
                        </div>
                      </div>
                      <button 
                        onClick={clearImage}
                        className="absolute top-2.5 right-2.5 p-1 bg-slate-950/90 text-slate-400 hover:text-white rounded-full border border-purple-900/30"
                        title="Remove image"
                      >
                        <X className="w-3.5 h-3.5" />
                      </button>
                    </div>
                  )}
                </div>

                {/* Processing Actions Grid */}
                <div className="grid grid-cols-1 sm:grid-cols-2 gap-3">
                  <button
                    onClick={generateSummary}
                    className="flex items-center justify-center space-x-2 py-3 px-4 bg-gradient-to-r from-purple-800 to-indigo-800 hover:from-purple-700 hover:to-indigo-700 active:scale-[0.98] transition-all rounded-xl text-xs font-semibold shadow-lg text-white group"
                  >
                    <FileText className="w-4 h-4 text-purple-300" />
                    <span>Generate Summary</span>
                  </button>
                  <button
                    onClick={generateNotes}
                    className="flex items-center justify-center space-x-2 py-3 px-4 bg-gradient-to-r from-purple-800 to-indigo-800 hover:from-purple-700 hover:to-indigo-700 active:scale-[0.98] transition-all rounded-xl text-xs font-semibold shadow-lg text-white group"
                  >
                    <Sparkles className="w-4 h-4 text-pink-300" />
                    <span>Forge Star-Notes</span>
                  </button>
                  <button
                    onClick={generateQuiz}
                    className="sm:col-span-2 flex items-center justify-center space-x-2 py-3.5 px-4 bg-gradient-to-r from-pink-600 to-purple-600 hover:from-pink-500 hover:to-purple-500 active:scale-[0.98] transition-all rounded-xl text-xs font-bold shadow-lg shadow-purple-900/20 text-white"
                  >
                    <Zap className="w-4 h-4 text-amber-300 animate-pulse" />
                    <span>Forge Quantum Quiz Challenge</span>
                  </button>
                </div>
              </div>

              {/* Chat Helper box */}
              <div className="bg-slate-900/60 border border-purple-900/40 rounded-2xl p-4 backdrop-blur-md shadow-xl flex flex-col h-[32rem]">
                <div className="flex items-center space-x-2 mb-3 px-2 border-b border-purple-950 pb-2">

                          return <h3 key={i} className="text-lg font-bold text-indigo-200 mt-5">{line.replace('##', '')}</h3>;

                          return <h2 key={i} className="text-xl font-extrabold text-transparent bg-clip-text bg-gradient-to-r from-pink-200 to-purple-200 border-b border-pink-900/40 pb-2 mb-4">{line.replace('#', '')}</h2>;
                        } else if (line.startsWith('-') || line.startsWith('*')) {
                          return <li key={i} className="ml-4 list-disc text-sm text-slate-300 py-0.5">{line.substring(2)}</li>;
                        } else if (line.trim() === '') {
                          return <div key={i} className="h-2" />;
                        } else {
                          return <p key={i} className="text-sm leading-relaxed text-slate-300">{line}</p>;

                  ) : (
                    <div className="h-full flex flex-col justify-center items-center text-center text-slate-500">

                              if (submittedQuiz) {

                                } else if (isSelected) {

                                >

                    <div className="absolute top-0 inset-x-0 h-1.5 bg-gradient-to-r from-purple-500 to-indigo-500" />
                    
                    <div className="max-w-md mx-auto space-y-4">

                      <div className="flex justify-center items-baseline space-x-1">

                        }

                      <div className="flex justify-center gap-3 pt-2">
                        <button
                          onClick={() => {
                            setSelectedAnswers({});
                            setSubmittedQuiz(false);
                            setQuizScore(0);
                            showToast("Quiz reset! Let's conquer it again.", "info");
                          }}
                          className="px-5 py-2.5 bg-slate-950 border border-purple-950 hover:bg-slate-900 text-slate-300 rounded-xl text-xs font-semibold"
                        >
                          Retry Exam
                        </button>
                        <button
                          onClick={() => {
                            setActiveTab('hub');
                          }}
                          className="px-5 py-2.5 bg-gradient-to-r from-purple-700 to-indigo-700 text-white hover:opacity-95 rounded-xl text-xs font-semibold"
                        >
                          Back to Intake Hub
                        </button>
                      </div>
                    </div>
                  </div>
                )}

              </div>
            )}

          </div>
        )}

        {/* TAB 3: NEBULA BOARD */}
        {activeTab === 'dreams' && (
          <div className="space-y-8">
            
            {/* Header description */}

              <div className="space-y-1">
                <div className="flex items-center space-x-2">
                  <Moon className="w-5 h-5 text-indigo-400" />
                  <h2 className="text-xl font-bold tracking-wide text-slate-200">Nebula Board</h2>

                </p>

              <div className="flex items-center space-x-4 bg-slate-950/60 p-3 rounded-xl border border-purple-950/60 text-xs">
                <div className="flex items-center space-x-1.5 text-indigo-300">
